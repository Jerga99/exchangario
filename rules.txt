rules_version = '2';
service cloud.firestore {
  match /databases/{database}/documents {

  	function isLoggedIn() {
    	return request.auth.uid != null;
    }

    match /users/{userId} {
    	allow read, write: if isLoggedIn();
    }

    match /exchanges/{exchangeId} {
    	allow read: if true;
      allow write: if isLoggedIn();
    }

    match /opportunities/{opportunityId} {
    	allow read, write: if isLoggedIn();
    }
  }
}
